28  /**
29   * An implementation of the Nominatim Api Service.
30   * 
31   * @author Jérémie Huchet
32   */
33  public final class JsonNominatimClient implements NominatimClient {
34  
35      /** The event logger. */
36      private static final Logger LOGGER = LoggerFactory.getLogger(JsonNominatimClient.class);
37  
38      /** The HTTP header "Accept". */
39      private static final String H_ACCEPT = "Accept";
40  
41      /** The url to make search queries. */
42      private final String searchUrl;
43  
44      /** The url to make a query for a reverse geocoding. */
45      private final String reverseGeocodingUrl;
46  
47      /** The HTTP client. */
48      private final HttpClient httpClient;
49  
50      /** The default response handler for search requests. */
51      private final NominatimSearchResponseHandler defaultSearchResponseHandler = new NominatimSearchResponseHandler();
52  
53      /** The default response handler for reverse geocoding requests. */
54      private final ReverseGeocodingResponseHandler defaultReverseGeocodingHandler = new ReverseGeocodingResponseHandler();
55  
56      /**
57       * Creates the json nominatim client.
58       * 
59       * @param httpClient
60       *            an HTTP client
61       * @param email
62       *            an email to add in the HTTP requests parameters to sign them
63       */
64      public JsonNominatimClient(final HttpClient httpClient, final String email) {
65  
66          this(httpClient, email, null, false);
67      }
68  
69      /**
70       * Creates the json nominatim client.
71       * 
72       * @param httpClient
73       *            an HTTP client
74       * @param email
75       *            an email to add in the HTTP requests parameters to sign them
76       * @param searchBounds
77       *            the prefered search bounds
78       * @param strictBounds
79       *            set to true if you want the results to be located into the given bounding box
80       */
81      public JsonNominatimClient(final HttpClient httpClient, final String email,
82              final BoundingBox searchBounds, final boolean strictBounds) {
83  
84          this.httpClient = httpClient;
85  
86          final StringBuilder searchUrlBuilder = new StringBuilder();
87          searchUrlBuilder
88                  .append("http://nominatim.openstreetmap.org/search?format=json&polygon=0&addressdetails=1&q=%s&email=");
89          searchUrlBuilder.append(email);
90  
91          if (searchBounds != null) {
92              searchUrlBuilder.append("&viewbox=");
93              searchUrlBuilder.append(toString(searchBounds.getWest())).append(",");
94              searchUrlBuilder.append(toString(searchBounds.getNorth())).append(",");
95              searchUrlBuilder.append(toString(searchBounds.getEast())).append(",");
96              searchUrlBuilder.append(toString(searchBounds.getSouth()));
97  
98              if (strictBounds) {
99                  searchUrlBuilder.append("&bounded=1");
100             }
101         }
102 
103         this.searchUrl = searchUrlBuilder.toString();
104         this.reverseGeocodingUrl = "http://nominatim.openstreetmap.org/reverse?format=json&polygon=0&addressdetails=1&lat=%s&lon=%s&email="
105                 + email;
106 
107         if (LOGGER.isDebugEnabled()) {
108             LOGGER.debug("API search URL: {}", searchUrl);
109             LOGGER.debug("API reverse geocoding URL: {}", reverseGeocodingUrl);
110         }
111     }
112 
113     /**
114      * {@inheritDoc}
115      * 
116      * @see fr.dudie.nominatim.client.NominatimClient#search(java.lang.String)
117      */
118     @Override
119     public List<Address> search(final String query) throws IOException {
120 
121         final String apiCall = String.format(searchUrl, URLEncoder.encode(query, "UTF-8"));
122         final HttpGet req = new HttpGet(apiCall);
123         req.addHeader(H_ACCEPT, "text/json");
124         req.addHeader(H_ACCEPT, "application/json");
125         req.addHeader(H_ACCEPT, "*/*");
126 
127         final List<Address> addresses = httpClient.execute(req, defaultSearchResponseHandler);
128         return addresses;
129     }
130 
131     /**
132      * {@inheritDoc}
133      * 
134      * @see fr.dudie.nominatim.client.NominatimClient#getAddress(double, double)
135      */
136     @Override
137     public final Address getAddress(final double longitude, final double latitude)
138             throws IOException {
139 
140         final String apiCall = String.format(reverseGeocodingUrl, toString(latitude),
141                 toString(longitude));
142         final HttpGet req = new HttpGet(apiCall);
143         req.addHeader(H_ACCEPT, "text/json");
144         req.addHeader(H_ACCEPT, "application/json");
145         req.addHeader(H_ACCEPT, "*/*");
146 
147         final Address address = httpClient.execute(req, defaultReverseGeocodingHandler);
148         return address;
149     }
150 
151     /**
152      * {@inheritDoc}
153      * 
154      * @see fr.dudie.nominatim.client.NominatimClient#getAddress(int, int)
155      */
156     @Override
157     public Address getAddress(final int longitudeE6, final int latitudeE6) throws IOException {
158 
159         return getAddress((longitudeE6 / 1E6), (latitudeE6 / 1E6));
160     }

162     /**
163      * Gets the string representation of a double value.
164      * 
165      * @param value
166      *            a double value
167      * @return the string representation (e. g. "48.054600")
168      */
169     private static String toString(final double value) {
170 
171         return String.format("%f", value).replaceAll(",", ".");
172     }
