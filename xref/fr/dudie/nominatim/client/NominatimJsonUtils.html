44      /**
45       * Converts a json string to a java object.
46       * 
47       * @param json
48       *            a json string representing an {@link Address}
49       * @return an address
50       * @throws JSONException
51       *             unable to parse json data
52       */
53      public static Address toAddress(final JSONObject json) throws JSONException {
54  
55          final Address address = new Address();
56          address.setOsmType(json.optString("osm_type"));
57  
58          final JSONArray jsonBbox = json.optJSONArray("boundingbox");
59          if (null != jsonBbox) {
60              address.setBoundingBox(getBoundingBox(jsonBbox));
61          }
62          if (LOGGER.isInfoEnabled() && null != json.optJSONArray("polygonpoints")) {
63              LOGGER.info("response conatins polygonpoints but they are ignored");
64          }
65  
66          address.setLatitude(json.optDouble("lat"));
67          address.setLongitude(json.optDouble("lon"));
68  
69          address.setDisplayName(json.optString("display_name"));
70          address.setElementClass(json.optString("class"));
71          address.setElementType(json.optString("type"));
72          address.setAddressElements(getAddressElements(json.optJSONObject("address")));
73  
74          return address;
75      }

77      /**
78       * Converts a json string representing a bounding box to a java object.
79       * 
80       * @param json
81       *            a json string containing an array of 4 double
82       * 
83       *            <pre>
84       *                 boundingbox: [
85       *                     "48.4065589904785"
86       *                     "48.4071006774902"
87       *                     "-4.52232980728149"
88       *                     "-4.52062273025513"
89       *                 ]
90       * </pre>
91       * @return a bounding box object
92       * @throws JSONException
93       *             unable to parse json data
94       */
95      private static BoundingBox getBoundingBox(final JSONArray json) throws JSONException {
96  
97          if (null == json) {
98              return null;
99          }
100 
101         if (json.length() != 4) {
102             final String msg = String
103                     .format("A json array representing a bounding box must contains 4 values (this one contains %s values)",
104                             json.length());
105             throw new JSONException(msg);
106         }
107         final BoundingBox bbox = new BoundingBox();
108         bbox.setEast(json.getDouble(0));
109         bbox.setWest(json.getDouble(1));
110         bbox.setNorth(json.getDouble(2));
111         bbox.setSouth(json.getDouble(3));
112         return bbox;
113     }
114 
115     /**
116      * Converts a json string representing an array of polygon points to a java object.
117      * 
118      * @param json
119      *            a json string containing an array of polygon points
120      * @return an array of polygon points
121      * @throws JSONException
122      *             unable to parse json data
123      */
124     private static PolygonPoint[] getPlygonPoints(final JSONArray json) throws JSONException {
125 
126         if (null == json) {
127             return new PolygonPoint[0];
128         }
129 
130         final PolygonPoint[] points = new PolygonPoint[json.length()];
131 
132         for (int i = 0; i < json.length(); i++) {
133             final JSONArray jsonPoint = json.getJSONArray(i);
134             final PolygonPoint p = new PolygonPoint();
135             p.setLongitude(jsonPoint.getDouble(0));
136             p.setLatitude(jsonPoint.getDouble(1));
137             points[i] = p;
138         }
139 
140         return points;
141     }

143     /**
144      * Extracts the address properties of a json object.
145      * 
146      * @param json
147      *            the json string containing the address properties
148      * @return the list of elements contained in the json string
149      * @throws JSONException
150      *             unable to parse json data
151      */
152     private static AddressElement[] getAddressElements(final JSONObject json) throws JSONException {
153 
154         if (null == json) {
155             return new AddressElement[0];
156         }
157 
158         final JSONArray keys = json.names();
159 
160         final AddressElement[] elems = new AddressElement[keys.length()];
161 
162         for (int i = 0; i < json.length(); i++) {
163             final AddressElement elem = new AddressElement();
164             elem.setKey(keys.getString(i));
165             elem.setValue(json.getString(elem.getKey()));
166             elems[i] = elem;
167         }
168         return elems;
169     }
